type arr : array[0:10] of int;

type date : struct {
    day : int;
    month : int;
    year : int;
} fstruct;

type details : struct {
    age : int;
    salary : float;
    birthday : date;
} fstruct;

var details : details;
var arr : arr;

details.age := 33;
details.salary := 1235.99;

details.birthday.day := 4;

details.birthday.month := 12;
details.birthday.year := 199;

arr[0] := 10;
details.birthday.year := arr[0] * details.birthday.year + details.age;
