PROG

type date : struct  {
    jour : int;
    mois : int;
    annee : float;
} fstruct;

func my_func() -> date {        // The function must return a value of a base type
    var x : date;
    return x;                   
}

proc my_proc(today : date) {    // Parameters should be of a base type
    var x : date;
    x := today;
}
